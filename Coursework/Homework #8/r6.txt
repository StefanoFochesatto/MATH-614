uNorms = []
vNorms = []
sNorms = []
aNorms = []
Acond = []

for i = 1:5
    [U, S, V] = svd(randn(50));
    S = diag(sort(rand(50,1).^6, 'descend'));
    A = U*S*V';
    [U2, S2, V2] = svd(A);
    Acond(i) = cond(A);
    V2 = V2*diag(round(diag(V'*V2)))';
    U2 = U2*diag(round(diag(U'*U2)))';

    uNorms(i) = norm(U - U2);
    vNorms(i) = norm(V - V2);
    sNorms(i) = norm(S - S2)/norm(S);
    aNorms(i) = norm(A - U2*S2*V2')/norm(A);
end
